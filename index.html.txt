<script>
let routines = JSON.parse(localStorage.getItem("routinesUI")) || [];
let timerInterval;

function save() {
    localStorage.setItem("routinesUI", JSON.stringify(routines));
}

function render() {
    const container = document.getElementById("routines");
    container.innerHTML = "";

    routines.forEach((routine, rIndex) => {
        let html = `
        <div class="card">
            <h3>${routine.name}</h3>

            <input placeholder="Ejercicio" id="name-${rIndex}">
            <input type="number" placeholder="Series" id="series-${rIndex}">
            <input type="number" placeholder="Reps" id="reps-${rIndex}">
            <input type="number" placeholder="Peso (kg)" id="weight-${rIndex}">
            <input type="number" placeholder="Descanso (seg)" id="rest-${rIndex}">
            <button class="add-btn" onclick="addExercise(${rIndex})">Agregar ejercicio</button>
        `;

        routine.exercises.forEach((ex, eIndex) => {
            html += `
            <div class="exercise">
                <strong>${ex.name}</strong><br>
                ${ex.series}x${ex.reps} | ${ex.weight}kg | ⏱ ${ex.rest}s
                <button class="add-btn" onclick="startTimer(${ex.rest})">Descanso</button>
                <button class="delete-btn" onclick="deleteExercise(${rIndex}, ${eIndex})">Eliminar</button>
            </div>
            `;
        });

        html += `<button class="delete-btn" onclick="deleteRoutine(${rIndex})">Eliminar rutina</button></div>`;
        container.innerHTML += html;
    });
}

function addRoutine() {
    const name = document.getElementById("routineName").value;
    if (!name) return alert("Escribe un nombre");

    routines.push({ name, exercises: [] });
    save();
    render();
    document.getElementById("routineName").value = "";
}

function addExercise(rIndex) {
    const name = document.getElementById(`name-${rIndex}`).value;
    const series = document.getElementById(`series-${rIndex}`).value;
    const reps = document.getElementById(`reps-${rIndex}`).value;
    const weight = document.getElementById(`weight-${rIndex}`).value;
    const rest = document.getElementById(`rest-${rIndex}`).value;

    if (!name || !series || !reps || !weight || !rest) {
        return alert("Completa todo");
    }

    routines[rIndex].exercises.push({ name, series, reps, weight, rest });
    save();
    render();
}

function deleteRoutine(index) {
    routines.splice(index, 1);
    save();
    render();
}

function deleteExercise(rIndex, eIndex) {
    routines[rIndex].exercises.splice(eIndex, 1);
    save();
    render();
}

function startTimer(seconds) {
    clearInterval(timerInterval);
    let time = seconds;
    const display = document.getElementById("timer");

    timerInterval = setInterval(() => {
        display.textContent = "⏱ " + time;

        if (time <= 0) {
            clearInterval(timerInterval);
            alert("Listo 💪");
        }
        time--;
    }, 1000);
}

render();
</script>
